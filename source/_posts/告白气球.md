- (void)centralManagerDidUpdateState:(CBCentralManager *)central {
    switch (central.state) {
        case CBManagerStateUnknown:
            label.text = @">>>CBManagerStateUnknown";
            break;
        case CBManagerStateResetting:
            label.text = @">>>CBManagerStateResetting";
            break;
        case CBManagerStateUnsupported:
            label.text = @">>>CBManagerStateUnsupported";
            break;
        case CBManagerStateUnauthorized:
            label.text = @">>>CBManagerStateUnauthorized";
            break;
        case CBManagerStatePoweredOff:
            label.text = @">>>CBManagerStatePoweredOff";
            break;
        default:
            label.text = @">>>CBManagerStatePoweredOn";
            // 开始扫描外设
            [manager scanForPeripheralsWithServices:nil options:nil];
            break;
    }
}
